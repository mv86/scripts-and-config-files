# Preferred theme
ZSH_THEME="flazz"

# Preferred editor for local and remote sessions
 if [[ -n $SSH_CONNECTION ]]; then
   export EDITOR='vi'
 else
   export EDITOR='vi'
 fi

# Compilation flags
export GOPATH=~/go
export GOWORKSPACE=$GOPATH/src/github.com/mv86

# Path exports
export PATH=$HOME/bin:$PATH
export PATH=$PATH:/usr/local/go/bin
export PATH=$PATH:$GOPATH/bin

# Program aliases
alias gch="google-chrome"
alias lo="libreoffice"
alias vb="virtualbox"
alias skype="skypeforlinux"

# Command line aliases
alias c="clear"
alias x="exit"
alias gphm="git push heroku master"
alias ni="npm install"
alias ns="npm start"

# Program language aliases
alias junit="java org.junit.runner.JUnitCore $1"
alias gowork="cd $GOWORKSPACE"

# Python aliases
# 'freeze' stops Debian bug (pkg-resourses==0.0.0) and pylint dependencies being included in requirements file
alias freeze="pip freeze | grep -vE 'pkg-resources==0.0.0|pylint|isort|astroid|mccabe' > requirements.txt"
alias actvenv="source venv/bin/activate"

# Zsh options
setopt histignorespace
